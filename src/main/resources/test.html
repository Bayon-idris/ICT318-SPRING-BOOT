function updateUe() { const ueId = document.getElementById('editUeId').value; const ueCode = document.getElementById('editUeCode').value; const ueTitle = document.getElementById('editUeTitle').value; const ueDescription = document.getElementById('editUeDescription').value; fetch(`/dashboard/update-ue/${ueId}`, { method: 'PUT', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ code: ueCode, title: ueTitle, description: ueDescription }) }) .then(response => { if (response.ok) { window.location.reload(); } else { console.error('Erreur lors de la mise à jour de l\'UE'); } }) .catch(error => { console.error('Erreur lors de la mise à jour de l\'UE', error); }); }
